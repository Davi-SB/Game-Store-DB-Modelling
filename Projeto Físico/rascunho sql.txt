Tabela Usuário(ID, Username!)
    ID é a PK

Tabela Amigo(Amigo1, Amigo2)
	Amigo1 -> Usuário(ID)
	Amigo2 -> Usuário(ID)
    (Amigo1, Amigo2) é a PK composta

Tabela Conta(ID, Senha!, Email!, Cartão)
	ID -> Usuário(ID)
    ID é a PK

Tabela Cartão(ID, Num, Cod, DT_Exp)
	ID -> Conta(ID)
    (ID, Num, Cod, DT_Exp) é a PK composta

Tabela Jogo(ID, Nome!, DT_Lançamento!, Preço!)
    ID é a PK

Tabela Comentário(Jogo_ID, Usuário_ID, Data, Estrelas!, Descrição)
	Jogo_ID -> Jogo(ID)
	Usuário_ID -> Usuário(ID)
    (Jogo_ID, Usuário_ID, Data) é a PK composta

Tabela Player(Usuário_ID) 
	Usuário_ID -> Usuário(ID)
    Usuário_ID é a PK

Tabela Compartilha(Jogo_ID, Dependente_ID, Dono_ID!)
	Jogo_ID -> Jogo(ID)
	Dependente_ID -> Player(ID)
	Dono_ID -> Player(ID)
    (Jogo_ID, Dependente_ID) é a PK

Tabela Tem(Usuário_ID, Jogo_ID)
	Jogo_ID -> Jogo(ID)
	Usuário_ID -> Usuário(ID)
    (Usuário_ID, Jogo_ID) é a PK composta

Tabela Conquistas(Usuário_ID, Jogo_ID, Conquista)
	(Usuário_ID, Jogo_ID) -> Tem(Usuário_ID, Jogo_ID)
    (Usuário_ID, Jogo_ID, Conquista) é a PK composta

Tabela Dev(Usuário_ID)
	Usuário_ID -> Usuário(ID)
    Usuário_ID é a PK

Tabela Empresa(ID, nome!)
    ID é a PK

Tabela Desenvolve(Usuário_ID, Jogo_ID, Empresa_ID)
	Jogo_ID -> Jogo(ID)
	Usuário_ID -> Usuário(ID)
	Empresa_ID -> Empresa(ID)
    (Usuário_ID, Jogo_ID) é a PK composta
